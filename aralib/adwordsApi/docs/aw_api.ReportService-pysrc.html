351      # Prepare XML for DOM construction. 
352      # 
353      # Report consists of two parts: table and totals. Table contains column 
354      # names and rows of data. The outgoing CSV will contain only columns and 
355      # data rows. The totals are not included, but can be easily calculated 
356      # from the CSV data. 
357      report_xml = report_xml.replace('<?xml version="1.0" standalone="yes"?>',

360      items = [] 
361      try: 
362        # Construct DOM object. 
363        dom = minidom.parseString(report_xml) 
364   
365        # Get data columns. 
366        columns = [] 
367        column_dom = dom.getElementsByTagName('column') 
368        for column_item in column_dom: 
369          if column_item.hasAttributes(): 
370            columns.append(column_item.attributes.item(0).value) 
371   
372        # Get data rows. 
373        rows = [] 
374        row_dom = dom.getElementsByTagName('row') 
375        for row_item in row_dom: 
376          if row_item.hasAttributes(): 
377            attrs = row_item.attributes 
378            row = {} 
379            for index in range(attrs.length): 
380              row[attrs.item(index).name] = attrs.item(index).value
